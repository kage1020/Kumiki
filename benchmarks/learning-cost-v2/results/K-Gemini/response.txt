# ====== Types ======

type CardId = nominal Text where uuid
type Column = Todo | Doing | Done
type Card   = {id: CardId, title: Text, column: Column, createdAt: Time}

# ====== Slots ======

slot cards : Map(CardId, Card) = {}
slot draft : Text              = ""
slot ready : Bool              = false

# ====== Effects ======

effect loadCards cap=storage.read
                 in=Unit
                 out=Result(Option(Map(CardId, Card)), Text)
                 policy=once
                 map-request={key: "kanban_cards", decode: Decoder.Json(Map(CardId, Card))}

effect saveCards cap=storage.write
                 in=Map(CardId, Card)
                 out=Result(Unit, Text)
                 policy=debounce(300ms)
                 map-request={key: "kanban_cards", value: $1}

# ====== Fns ======

fn nextColumn(c: Column) -> Column
   = match c with
       | Todo  -> Doing
       | Doing -> Done
       | Done  -> Done

fn prevColumn(c: Column) -> Column
   = match c with
       | Done  -> Doing
       | Doing -> Todo
       | Todo  -> Todo

fn canMoveLeft(c: Column) -> Bool
   = match c with
       | Todo  -> false
       | Doing -> true
       | Done  -> true

fn canMoveRight(c: Column) -> Bool
   = match c with
       | Todo  -> true
       | Doing -> true
       | Done  -> false

fn cardsInColumn(cs: Map(CardId, Card), c: Column) -> List(CardId)
   = cs.keys.filter(cs[$1].column == c).sort-by(cs[$1].createdAt.to-ms)

fn countInColumn(cs: Map(CardId, Card), c: Column) -> Int
   = cardsInColumn(cs, c).length

# ====== Reducers ======

reducer boot
    on=app.start
    do= emit loadCards()

reducer cardsLoaded
    on=loadCards.ok($m, _)
    do= cards := $m.get-or({})
        ready := true

reducer cardsFailed
    on=loadCards.err(_, _)
    do= ready := true

reducer addCard
    on=ui.submit(AddForm)
    do= let id = CardId.fresh()
        cards[id] := {id, title=draft, column=Todo, createdAt=now}
        draft := ""
        emit saveCards(cards)

reducer moveLeft
    on=ui.click(MoveLeftBtn)
    do= cards[$el.cardId].column := prevColumn(cards[$el.cardId].column)
        emit saveCards(cards)

reducer moveRight
    on=ui.click(MoveRightBtn)
    do= cards[$el.cardId].column := nextColumn(cards[$el.cardId].column)
        emit saveCards(cards)

reducer deleteCard
    on=ui.click(DeleteBtn)
    do= cards := cards.remove($el.cardId)
        emit saveCards(cards)

# ====== Tiles ======

tile MoveLeftBtn
    in=Bool
    = button(text="←", onClick=moveLeft, disabled=$1) { variant: "ghost" }

tile MoveRightBtn
    in=Bool
    = button(text="→", onClick=moveRight, disabled=$1) { variant: "ghost" }

tile DeleteBtn
    = button(text="✕", onClick=deleteCard) { variant: "ghost", color: "danger" }

tile CardView
    in=CardId
    = box(
        column(
          text(cards[$1].title) {weight: "bold"},
          row(
            row(
              MoveLeftBtn(not canMoveLeft(cards[$1].column)) {cardId: $1},
              MoveRightBtn(not canMoveRight(cards[$1].column)) {cardId: $1}
            ) {gap: "xs"},
            DeleteBtn {cardId: $1}
          ) {justify: "between", align: "center"}
        ) {gap: "sm"}
      ) {pad: "sm", shadow: "sm", bg: "surface", radius: "md"}

tile ColumnView
    in=Column
    = box(
        column(
          heading($1.show + " (" + countInColumn(cards, $1).show + ")") {size: "lg"},
          for id in cardsInColumn(cards, $1)
            CardView(id) {key: id.show}
        ) {gap: "md", align: "stretch"}
      ) {pad: "md", bg: "bg", radius: "md"}

tile AddForm
    = form(
        row(
          input(bind=draft, placeholder="New task...", auto-focus=true) {w: "full"},
          button(text="Add", type="submit", disabled=draft.is-empty) {bg: "primary", color: "surface"}
        ) {gap: "sm", align: "center"}
      )

tile Board
    = grid(
        ColumnView(Todo),
        ColumnView(Doing),
        ColumnView(Done)
      ) {cols: 3, gap: "md"}

tile App
    = page(
        heading("Kanban Board") {size: "xxl"},
        AddForm,
        when(ready, Board),
        when(not ready, spinner() {size: "lg"})
      ) {pad: "lg", gap: "lg", max-w: 1200}

tile NotFound = page(
                  heading("404"),
                  link(to="/") {text: "Back"})
